41 int MwInit(char *cmdBuf, uint16_t bufLen) {
42  // Check input params
43  if ((cmdBuf == NULL) || (bufLen < MW_CMD_MIN_BUFLEN)) return MW_ERROR;
44 
45  // Set command buffer
46  cmd = (MwCmd*)cmdBuf;
47  maxLen = bufLen;
48 
49  // Initialize LSD
50  LsdInit();
51 
52  // TODO Set lines to default status (keep WiFi module in reset)
53  MwModuleReset();
54  // Power down and Program not active (required for the module to boot)
55  UartClrBits(MCR, MW__PRG | MW__PD);
56  // Try accessing UART scratch pad register to see if it is installed
57 
58  UART_SPR = 0x55;
59  if (UART_SPR != 0x55) return MW_ERROR;
60  UART_SPR = 0xAA;
61  if (UART_SPR != 0xAA) return MW_ERROR;
62 
63  // Clear data received flag
64  fDr = FALSE;
65  recvd = 0;
66  // Enable control channel
67  LsdChEnable(MW_CTRL_CH);
68 
69  mwReady = TRUE;
70 
71  return MW_OK;
72 }

506 /************************************************************************/
517 int MwRecv(uint8_t **data, uint16_t *len, uint32_t maxLoopCnt) {
518  int ch;
519 
520  if (!mwReady) return MW_ERROR;
521 
522  // If data already received and pending, return immediately
523  if (fDr) {
524  ch = fDr;
525  fDr = FALSE;
526  *data = (uint8_t*)cmd;
527  *len = recvd;
528  recvd = 0;
529  return ch;
530  }
531 
532  *len = maxLen;
533  ch = LsdRecv((uint8_t*)cmd, len, maxLoopCnt);
534 
535  // Check if data received on control channel
536  if (0 == ch) {
537  *data = NULL;
538  *len = 0;
539  return 0;
540  }
541 
542  // Check if error
543  if (ch > MW_MAX_SOCK) {
544  *data = NULL;
545  *len = 0;
546  return MW_ERROR;
547  }
548 
549  // Assign received data and return
550  *data = (uint8_t*)cmd;
551 
552  return ch;
553 }
