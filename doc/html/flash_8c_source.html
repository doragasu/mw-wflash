55 uint8_t FlashDataPoll(uint32_t addr, uint8_t data) {
56  uint8_t read;
57 
58  // Poll while DQ7 != data(7) and DQ5 == 0 and DQ1 == 0
59  do {
60  read = FlashRead(addr);
61  } while (((data ^ read) & 0x80) && ((read & 0x22) == 0));
62 
63  // DQ7 must be tested after another read, according to datasheet
64 // if (((data ^ read) & 0x80) == 0) return 0;
65  read = FlashRead(addr);
66  if (((data ^ read) & 0x80) == 0) return 0;
67  // Data has not been programmed, return with error. If DQ5 is set, also
68  // issue a reset command to return to array read mode
69  if (read & 0x20) {
70  FlashReset();
71  }
72  // If DQ1 is set, issue the write-to-buffer-abort-reset command.
73  if (read & 0x02) {
74  FlashUnlock();
75  FlashReset();
76  }
77  return 1;
78 }

176 uint8_t FlashWriteBuf(uint32_t addr, uint16_t data[], uint8_t wLen) {
177  // Sector address
178  uint32_t sa;
179  // Number of words to write
180  uint8_t wc;
181  // Index
182  uint8_t i;
183 
184  // Check maximum write length
185  if (wLen > FLASH_CHIP_WBUFLEN) {
186  return 0;
187  }
188 
189  // Obtain the sector address
190  sa = FLASH_SA_GET(addr);
191  // Compute the number of words to write minus 1. Maximum number is 15,
192  // but without crossing a write-buffer page
193  wc = MIN(wLen, FLASH_CHIP_WBUFLEN - ((addr>>1) &
194  (FLASH_CHIP_WBUFLEN - 1))) - 1;
195  // Unlock and send Write to Buffer command
196  FlashUnlock();
197  FlashWriteW(sa, FLASH_WR_BUF[0]);
198  // Write word count - 1
199  FlashWriteW(sa, wc);
200 
201  // Write data to bufffer
202  for (i = 0; i <= wc; i++, addr+=2) FlashWriteW(addr, data[i]);
203  // Write program buffer command
204  FlashWriteW(sa, FLASH_PRG_BUF[0]);
205  // Poll until programming is complete
206  if (FlashDataPoll(addr - 1, data[i - 1] & 0xFF)) {
207  return 0;
208  }
209 
210  // Return number of elements (words) written
211  return i;
212 }

289 uint8_t FlashRangeErase(uint32_t addr, uint32_t len) {
290  // Index
291  uint8_t i, j;
292  // Shifted address to compare with
293  uint16_t caddr = addr>>FLASH_SADDR_SHIFT;
294  uint16_t clen = (len - 1)>>FLASH_SADDR_SHIFT;
295 
296  if (!len) return 0;
297  if ((addr + len) >= (FLASH_SADDR_MAX<<FLASH_SADDR_SHIFT)) return 1;
298 
299  // Find sector containing the initial address
300  for (i = FLASH_NSECT - 1; caddr < saddr[i]; i--);
301  // Find sector containing the end address
302  for (j = FLASH_NSECT - 1; (caddr + clen) < saddr[j]; j--);
303 
304  for (; i <= j; i++) {
305  if (!FlashSectErase((saddr[i])<<FLASH_SADDR_SHIFT)) return 2;
306  }
307 
308  return 0;
309 }
