48 int SfWrite(WfBuf *in, uint16_t len) {
49  uint16_t i, wlen;
50  uint8_t wwrite;
51 
52  // Convert byte length into word length
53  wlen = len>>1;
54 
55  if (wlen > d.wrem) {
56  return -1;
57  }
58 
59  // Write not aligned data
60  wwrite = FLASH_CHIP_WBUFLEN - (d.waddr & (FLASH_CHIP_WBUFLEN - 1));
61  wwrite = MIN(wwrite, wlen);
62  if (FlashWriteBuf(d.waddr<<1, in->wdata, wwrite) != wwrite) return -1;
63  d.waddr += wwrite;
64  // Write received buffer in blocks
65  for (i = wwrite; i < wlen; i += wwrite, d.waddr += wwrite) {
66  wwrite = MIN(FLASH_CHIP_WBUFLEN, wlen - i);
67  if (FlashWriteBuf(d.waddr<<1, in->wdata + i, wwrite) != wwrite) {
68  return -1;
69  }
70 
71  }
72  // Update remaining words, and check if finished
73  d.wrem -= wlen;
74  if (!d.wrem) return 1;
75  return 0;
76 }

83 int SfCycle(void) {
84  int retVal = 0;
85  WfBuf *in;
86  uint16_t len, lenTmp;
87  int ch;
88  int i;
89  uint32_t entry;
90 
91  // Receive data
92  if ((ch = MwRecv((uint8_t**)&in, &len, MW_DEF_MAX_LOOP_CNT)) == MW_ERROR)
93  return MW_ERROR;
94  if (ch != WF_CHANNEL) return MW_OK; // No data has been received
95 
96  // If we have received no payload, maybe the connection has been lost
97  if (!len) {
98  // Check socket status
99  if (MW_SOCK_TCP_EST != MwSockStatGet(WF_CHANNEL)) return MW_ERROR;
100  else return MW_OK;
101  }
102 
103  // While there is data to process...
104  while (len) {
105  // Process received data depending on state
106  switch (d.s) {
107  case WF_IDLE:
108  // Awaiting command
109  // NOTE: A write command should never be accompanied by data
110  // payload, write command must be acknowledged befor client
111  // starts sending data.
112  switch (ByteSwapWord(in->cmd.cmd)) {
113  // Get bootloader version
114  case WF_CMD_VERSION_GET:
115  // sanity checks
116  if (0 == ByteSwapWord(in->cmd.len) &&
117  WF_HEADLEN == len) {
118  in->cmd.len = ByteSwapWord(2);
119  in->cmd.cmd = WF_CMD_OK;
120  in->cmd.data[0] = WF_VERSION_MAJOR;
121  in->cmd.data[1] = WF_VERSION_MINOR;
122  MwSend(WF_CHANNEL, in->data, WF_HEADLEN + 2);
123  } else {
124  in->cmd.len = 0;
125  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
126  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
127  }
128  break;
129 
130  // Get Flash IDs
131  case WF_CMD_ID_GET:
132  // sanity checks
133  if (0 == ByteSwapWord(in->cmd.len) &&
134  WF_HEADLEN == len) {
135  in->cmd.data[0] = FlashGetManId();
136  FlashGetDevId(in->cmd.data + 1);
137  for (i = 0; i < 4; i++) in->cmd.data[i] =
138  in->cmd.data[i];
139  in->cmd.cmd = WF_CMD_OK;
140  in->cmd.len = ByteSwapWord(4);
141  MwSend(WF_CHANNEL, in->data, WF_HEADLEN + 4);
142  } else {
143  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
144  in->cmd.len = 0;
145  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
146  }
147  break;
148 
149  // Echo data (for debugging purposes)
150  case WF_CMD_ECHO:
151  // sanity check
152  if (len == ByteSwapWord(in->cmd.len) + WF_HEADLEN) {
153  in->cmd.cmd = WF_CMD_OK;
154  MwSend(WF_CHANNEL, in->data, len);
155  } else {
156  in->cmd.len = 0;
157  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
158  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
159  }
160  break;
161 
162  // Erase flash
163  case WF_CMD_ERASE:
164  // sanity check
165  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
166  if (((sizeof(WfMemRange) + WF_HEADLEN) == len) &&
167  (sizeof(WfMemRange) == ByteSwapWord(in->cmd.len))) {
168  VdpLineClear(VDP_PLANEA_ADDR, 3);
169  VdpDrawText(VDP_PLANEA_ADDR, 1, 3,
170  VDP_TXT_COL_MAGENTA, "ERASING...");
171  if (!FlashRangeErase(ByteSwapDWord(
172  in->cmd.mem.addr), ByteSwapDWord(
173  in->cmd.mem.len))) {
174  in->cmd.cmd = WF_CMD_OK;
175  }
176  }
177  VdpLineClear(VDP_PLANEA_ADDR, 3);
178  in->cmd.len = 0;
179  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
180  break;
181 
182  // Program flash
183  case WF_CMD_PROGRAM:
184  // sanity check
185  if ((len == (ByteSwapWord(in->cmd.len) + WF_HEADLEN)) &&
186  ((ByteSwapDWord(in->cmd.mem.addr) +
187  ByteSwapDWord(in->cmd.mem.len)) <
188  FLASH_CHIP_LENGTH)) {
189  // Acknowledge command and transition to
190  // WF_DATA_WAIT
191  VdpDrawText(VDP_PLANEA_ADDR, 1, 3,
192  VDP_TXT_COL_MAGENTA, "PROGRAM:");
193  VdpDrawHex(VDP_PLANEA_ADDR, 10, 3,
194  VDP_TXT_COL_MAGENTA, in->cmd.data[2]);
195  VdpDrawHex(VDP_PLANEA_ADDR, 12, 3,
196  VDP_TXT_COL_MAGENTA, in->cmd.data[1]);
197  VdpDrawHex(VDP_PLANEA_ADDR, 14, 3,
198  VDP_TXT_COL_MAGENTA, in->cmd.data[0]);
199  in->cmd.cmd = WF_CMD_OK;
200  d.s = WF_DATA_WAIT;
201  d.waddr = ByteSwapDWord(in->cmd.mem.addr)>>1;
202  d.wrem = ByteSwapDWord(in->cmd.mem.len)>>1;
203  } else in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
204  in->cmd.len = 0;
205  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
206  break;
207 
208  // Run program from address
209  case WF_CMD_RUN:
210  in->cmd.len = ByteSwapWord(0);
211  // sanity check
212  if (((WF_HEADLEN + 4) == len) &&
213  (4 == ByteSwapWord(in->cmd.len))) {
214  // Boot program at specified address
215  entry = ByteSwapDWord(in->cmd.dwdata[0]);
216  in->cmd.cmd = WF_CMD_OK;
217  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
218  SfBoot(entry);
219  } else {
220  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
221  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
222  }
223  break;
224 
225  case WF_CMD_AUTORUN:
226  in->cmd.len = ByteSwapWord(0);
227  // sanity check
228  if ((WF_HEADLEN == len) &&
229  (0 == ByteSwapWord(in->cmd.len))) {
230  in->cmd.cmd = WF_CMD_OK;
231  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
232  SfBoot(SF_ENTRY_POINT_ADDR);
233  } else {
234  in->cmd.cmd = ByteSwapWord(WF_CMD_ERROR);
235  MwSend(WF_CHANNEL, in->data, WF_HEADLEN);
236  }
237  break;
238 
239  default:
240  break;
241  }
242  // All the commands above either use all data, or error, so
243  // no more data to process here.
244  len = 0;
245  break;
246 
247  case WF_DATA_WAIT:
248  // Because command does not require confirmation at this point,
249  // if we receive more data than remaining to write, it could be
250  // because TCP has joined on a single datagram the data payload
251  // and a command request sent immediately after the data. So
252  // there is more data to process after we finish writing.
253  //
254  // TODO: It could happen that what is received after the data
255  // payload is a partial command (i.e. some command data is
256  // still missing). Support for handling this case is not yet
257  // implemented.
258  lenTmp = MIN(len, d.wrem<<1);
259  len -= lenTmp;
260  // Write data to Flash
261  switch (SfWrite(in, lenTmp)) {
262  case -1: // Error
263  VdpDrawText(VDP_PLANEA_ADDR, 1, 4, VDP_TXT_COL_MAGENTA,
264  "ERROR!");
265 
266  case 1: // Finished
267  d.s = WF_IDLE;
268  VdpLineClear(VDP_PLANEA_ADDR, 3);
269  break;
270 
271  case 0: // OK, more data pending
272  default:
273  break;
274  }
275  in = (WfBuf*)(in->data + lenTmp);
276  break;
277 
278  default:
279  break;
280  }// switch (d.s)
281  }// while (len)
282  return retVal;
283 }
