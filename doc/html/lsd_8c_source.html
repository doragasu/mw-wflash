48 static inline int LsdPollSend(uint8_t data[], uint16_t len,
49  uint32_t maxLoopCnt) {
50  int16_t i;
51  uint8_t n;
52  uint32_t loopCnt = maxLoopCnt;
53 
54  // Buffer is sent in chunks of up to UART_TX_FIFO_LEN bytes.
55  // Only the last chunck gets some advantage of the FIFOs :(
56  for (i = 0; i < len;) {
57  n = MIN(UART_TX_FIFO_LEN, len - i);
58  while (!UartTxReady() && loopCnt) loopCnt--;
59  if (!maxLoopCnt) return -1;
60  while (n--) UartPutc(data[i++]);
61  loopCnt = maxLoopCnt;
62  }
63  return len;
64 }

131 int LsdSend(uint8_t *data, uint16_t len, uint8_t ch, uint32_t maxLoopCnt) {
132  if (ch >= LSD_MAX_CH) {
133  return -1;
134  }
135  if (len > LSD_MAX_LEN) {
136  return -1;
137  }
138  if (!d.en[ch]) {
139  return -1;
140  }
141 
142  while ((!UartTxReady()) && maxLoopCnt) maxLoopCnt--;
143  if (!maxLoopCnt) return 0;
144  // Send STX
145  UartPutc(LSD_STX_ETX);
146  // Send ch and high nibble of length
147  UartPutc((ch<<4) | (len>>8));
148  // Send low byte of length
149  UartPutc(len & 0xFF);
150  // Send data payload
151  if (LsdPollSend(data, len, UINT32_MAX) != len) return -1;
152  // Send ETX
153  // TODO: It's stupid using FIFOs to end up doing this:

155  if (!maxLoopCnt) return -1;
156  UartPutc(LSD_STX_ETX);

256 int LsdRecv(uint8_t* buf, uint16_t* maxLen, uint32_t maxLoopCnt) {
257  LsdState rxs = LSD_ST_STX_WAIT;
258  uint32_t loops;
259  uint16_t pos = 0;
260  uint16_t len = 0;
261  uint8_t recv;
262  int8_t ch = -1;
263 
264  while (1) {
265  // Receive a character
266  loops = maxLoopCnt;
267  while (!UartRxReady()) {
268  loops--;
269  if (!loops) return -1;
270  }
271  recv = UartGetc();
272  switch (rxs) {
273  case LSD_ST_IDLE: // Do nothing!
274  return -1;
275 
276  case LSD_ST_STX_WAIT: // Wait for STX to arrive
277  if (LSD_STX_ETX == recv) rxs = LSD_ST_CH_LENH_RECV;
278  break;
279 
280  case LSD_ST_CH_LENH_RECV: // Receive CH and len high
281  // Check special case: if we receive STX and pos == 0,
282  // then this is the real STX (previous one was ETX from
283  // previous frame!).
284  if (!(LSD_STX_ETX == recv && 0 == pos)) {
285  ch = recv>>4;
286  len = (recv & 0x0F)<<8;
287  // Sanity check (not exceding number of channels)
288  if (ch >= LSD_MAX_CH) {
289  return -1;
290  }
291  else rxs = LSD_ST_LEN_RECV;
292  }
293  break;
294 
295  case LSD_ST_LEN_RECV: // Receive len low
296  len |= recv;
297  // Sanity check (not exceeding maximum buffer length)
298  if (len > *maxLen) {
299  return -1;
300  }
301  // If there's payload, receive it. Else wait for ETX
302  if (len) {
303  pos = 0;
304  rxs = LSD_ST_DATA_RECV;
305  } else {
306  rxs = LSD_ST_ETX_RECV;
307  }
308  break;
309 
310  case LSD_ST_DATA_RECV: // Receive payload
311  buf[pos++] = recv;
312  if (pos >= len) rxs = LSD_ST_ETX_RECV;
313  break;
314 
315  case LSD_ST_ETX_RECV: // ETX should come here
316  if (LSD_STX_ETX == recv) {
317  *maxLen = pos;
318  return ch;
319  }
320  // Error, ETX not received.
321  return -1;
322 
323  default:
324  // Code should never reach here!
325  return -1;
326  } // switch(d.rxs)
327  } // while (1)
