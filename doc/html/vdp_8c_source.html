47 void VdpInit(void) {
48  // Clear VRAM
49  VdpVRamClear(0, 32768);
50  // Mode 1:
51  // - No 8 left pixels blank
52  // - No HINT
53  VdpRegWrite(VDP_REG_MODE1, 0x04);
54  // Mode 2:
55  // - Display disabled
56  // - VBlank int disabled
57  // - DMA disabled
58  // - 224 lines
59  // - Megadrive mode
60  VdpRegWrite(VDP_REG_MODE2, 0x04);
61  // Name table for PLANE A set to 0x2000
62  VdpRegWrite(VDP_REG_PLANEA_NT, VDP_PLANEA_ADDR>>10);
63  // Name table for WINDOW set to 0x6000
64  VdpRegWrite(VDP_REG_WIN_NT, VDP_WIN_ADDR>>10);
65  // Name table for PLANE B set to 0x4000
66  VdpRegWrite(VDP_REG_PLANEB_NT, VDP_PLANEB_ADDR>>13);
67  // Empty sprite attribute table
68  VdpRegWrite(VDP_REG_SPR_T, 0x00);
69  // Background color palette 0, color 0
70  VdpRegWrite(VDP_REG_BGCOL, 0x00);
71  // Mode 3:
72  // - External interrupt disable
73  // - No vertical tile scroll
74  // - Plane horizontal scroll
75  VdpRegWrite(VDP_REG_MODE3, 0x00);
76  // Mode 4:
77  // - 40 tiles per line
78  // - VSync signal
79  // - HSync signal
80  // - Standard color data generation
81  // - No shadow/highlight mode
82  // - No interlace
83  VdpRegWrite(VDP_REG_MODE4, 0x81);
84  // Horizontal scroll data address
85  VdpRegWrite(VDP_REG_HSCROLL, 0x00);
86  // Set auto-increment to 2
87  VdpRegWrite(VDP_REG_INCR, 0x02);
88  // Set plane sizes to 64x64 cells
89  VdpRegWrite(VDP_REG_PSIZE, 0x11);
90  // Window H and V positions
91  VdpRegWrite(VDP_REG_WIN_HPOS, 0x00);
92  VdpRegWrite(VDP_REG_WIN_VPOS, 0x00);
93  // Load font three times, to be able to use three different colors
94  VdpFontLoad(font, fontChars, 0, 1, 0);
95  VdpFontLoad(font, fontChars, fontChars * 32, 2, 0);
96  VdpFontLoad(font, fontChars, 2 * fontChars * 32, 3, 0);
97 
98  // Set background and font colors
99  VdpRamRwPrep(VDP_CRAM_WR, 0);
100  VDP_DATA_PORT_W = VDP_COLOR_BLACK;
101  VDP_DATA_PORT_W = VDP_COLOR_WHITE;
102  VDP_DATA_PORT_W = VDP_COLOR_CYAN;
103  VDP_DATA_PORT_W = VDP_COLOR_MAGENTA;
104 
105  // Set vertical scroll to 0
106  VDP_DATA_PORT_W = 0;
107  VdpRamRwPrep(VDP_VSRAM_WR, 0);
108  VDP_DATA_PORT_W = 0;
109  VDP_DATA_PORT_W = 0;
110 
111  // Enable display
112  VdpRegWrite(VDP_REG_MODE2, 0x44);
113 }

188 uint8_t VdpDrawDec(uint16_t planeAddr, uint8_t x, uint8_t y, uint8_t txtColor,
189  uint8_t num) {
190  uint16_t offset;
191  uint8_t tmp, i = 0;
192 
193  // Calculate nametable offset and prepare VRAM writes
194  offset = planeAddr + 2 * (x + y * 64);
195  VdpRamRwPrep(VDP_VRAM_WR, offset);
196 
197  // Compute digits and write decimal number
198  // On 3 digit numbers, first one can only be 1 or 2. Take advantage of
199  // this to avoid division (TODO test if this is really faster).
200  if (num > 199) {
201  VDP_DATA_PORT_W = txtColor + 0x10 + 2;
202  num -= 200;
203  i++;
204  } else if (num > 99) {
205  VDP_DATA_PORT_W = txtColor + 0x10 + 1;
206  num -= 100;
207  i++;
208  }
209  tmp = num / 10;
210  num %= 10;
211 
212  if (tmp || i) {
213  VDP_DATA_PORT_W = txtColor + 0x10 + tmp;
214  i++;
215  }
216  VDP_DATA_PORT_W = txtColor + 0x10 + num;
217  i++;
218 
219  return i;
220 }

232 void VdpFontLoad(const uint32_t font[], uint8_t chars, uint16_t addr,
233  uint8_t fgcol, uint8_t bgcol) {
234  uint32_t line;
235  uint32_t scratch;
236  int16_t i;
237  int8_t j, k;
238 
239  // Prepare write
240  VdpRamRwPrep(VDP_VRAM_WR, addr);
241  // Note font is 1bpp, and it expanded to 4bpp during load
242 
243  // Each char takes two DWORDs
244  for (i = 0; i < 2 * chars; i++) {
245  line = font[i];
246  // Process a 32-bit font input
247  for (j = 0; j < 4; j++) {
248  scratch = 0;
249  // Process 8 bits to create 1 VDP DWORD
250  for (k = 0; k < 8; k++) {
251  scratch<<=4;
252  scratch |= line & 1?fgcol:bgcol;
253  line>>=1;
254  }
255  // Write 32-bit value to VDP
256  VDP_DATA_PORT_DW = scratch;
257  }
258  }
259 }
